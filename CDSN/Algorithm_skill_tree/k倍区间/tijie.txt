 
#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    long long n, k, ans = 0, son[100000], sum[100000], b[100000] = {0};//son是用来储存数的，sum是用来储存前x项的余数和的，b是用来储存相同余数的区间的个数的
    cin >> n >> k;
    for (int i = 0; i < n; i++)
    {
        cin >> son[i];
        if (i != 0)
           sum[i] = (sum[i - 1] + son[i]) % k;//如果不是第一个数，输入的数与前面残留余数的和除余数等于新的前i项和的余数
        else
            sum[i] = son[i] % k;
        b[sum[i]]++;//b[k]表示余数为k的区间个数，b[0]=2,b[1]=3
        ans += b[sum[i]] - 1;//假设现在有4个区间余数都为a，那么K倍区间有1+2+3=6个，这里只表示了区间之间的关系，没有表示前i项和恰好为k的倍数的情况
        //现在有2个区间余数为0，k倍区间有1个；3个区间余数为1，k倍区间有1+2=3个
        if (sum[i] == 0)//这里表示的就是前i项和恰为k的倍数的情况
            ans++;
    }
    cout << ans;
    return 0;
}
