void DeleteRepeat(int &A[],i,j,k[],t)//k[]存放被删除的元素
{
    if(A[]==null)
    return false;
    else
    {
        for(i=0,j=A.length-1;i<j;i++)
        {
            if(A[i]!=A[j])
            j--;
            else{
                k[t]=A[i];
                t++;
                A[i]=A[i+1];
            }
        }
    }
}
